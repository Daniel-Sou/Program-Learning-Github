Format II.

``` cpp
#include <stdlib.h>
#include <iostream>

using namespace std;

int main(){
    int i,a = 10;

    // newNum = 10 + (i = i + 1);
    int newNum = 10 + ++i;

    // ++a = 10 + 1 = 11;
    cout << ++a << "\n";
    cout << "Result is " << newNum << endl;
}
```
